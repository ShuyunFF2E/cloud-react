import React from "react";
import { Button, InputNumber } from "cloud-react";

const blank = "\u00A0";

class ButtonDemo extends React.Component {
  constructor(props) {
    super(props);

    this.state = {
      loading: false,
    };
  }

  handlerCLick = () => {
    this.setState({ loading: true }, () => {
      setTimeout(() => {
        this.setState({ loading: false });
      }, 5000);
    });
  };

  handlerCompactCLick = async () => {
    await new Promise((resolve) => {
      setTimeout(() => {
        resolve();
      }, 5000);
    });
  };

  render() {
    const { loading } = this.state;
    const text = loading ? "加载中…" : "点击后变为加载中";
    return (
      <ul>
        <li style={{ marginBottom: 10 }}>
          <h5>异步方法回调：</h5>
          <div style={{ display: "flex", alignItems: "center" }}>
            <Button
              style={{ marginRight: 5 }}
              onClick={this.handlerCompactCLick}
            >
              请点击
            </Button>
            <Button.Group>
              <Button onClick={this.handlerCompactCLick}>上一页</Button>
              <Button onClick={this.handlerCompactCLick}>下一页</Button>
            </Button.Group>
          </div>
        </li>
        <li style={{ marginBottom: 10 }}>
          <h5>主要：</h5>

            style={{ marginRight: 5 }}

            loading={this.state.loading}
            onClick={this.handlerCLick}
          >
            {text}
