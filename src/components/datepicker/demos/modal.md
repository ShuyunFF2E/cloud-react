import React from "react";
import { Button, Datepicker, Modal } from "cloud-react";

class DatepickerDemo extends React.Component {

    super(props);

      visible: false,
    };

  openBasicModal = () => {
    this.setState({
      visible: true,
    });

    this.setState({
      visible: false,
    });
  };

  render() {
    return (
      <div>
        <Button type="primary" onClick={this.openBasicModal}>
          基本弹出框

            showTimePicker={true}
            onChange={this.onInpChange}
